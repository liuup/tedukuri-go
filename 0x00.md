# 位运算

## [AcWing 89 a^b](https://www.acwing.com/problem/content/91/)

```
// 乘法快速幂
package main

import (
    "fmt"
)

func main() {
    var a, b, p int64
    
    fmt.Scanf("%d %d %d", &a, &b, &p)
    
    ans := qpow_for_mod(a, b, p)
    
    fmt.Println(ans)
}

// 循环快速幂 取模
func qpow_for_mod(a, n, mod int64) int64 {
	ans := int64(1 % mod)
	for n > 0 {
		if n&1 == 1 { // n的末位为1，用来判断奇偶性
			ans = int64(ans * a % mod)
		}
		a = int64(a * a % mod)
		n >>= 1
	}
	return ans
}
```
